Docker Client Installation => cd /usr/ports/sysutils/docker => sudo make install clean # Conflicts with the server installation # Looks like the Docker Server installs the Client too. Server Installation # You need to install this package: # To enables SSL Certificate Verification => sudo pkg install -y ca_root_nss # Before installing the Docker you need to setup # the dataset for it. => sudo zfs create zroot/usr/docker # Mount it to “/usr/docker” => sudo zfs set mountpoint=/usr/docker zroot/usr/docker # Note: You could have done above two commands with # this single line: # => sudo zfs create -o mountpoint=/usr/docker zroot/docker # Installing the Docker server => cd /usr/ports/sysutils/docker-freebsd => sudo make install clean # Enable the Docker service in the “/etc/rc.conf" => sudo sysrc -f /etc/rc.conf docker_enable="YES" => sudo service docker start
