The difference between “++i” and “i++” #include <stdio.h> /* If you use the "++i" (pre-increment) or "i++" (post-increment) * just itself ( i.e. without any asignment, there is no difference ! * "++i" = "i++" = i + 1 * However, you asign a value ( n = ++i or m = i++), then : * "++i" will return the current value of "i" + "1" immidiatly ! * "i++" will return the current value of "i" first, then adds "1" to the current value * Next time the current will be "i" + "1" */ int main( int argc, char* argv[] ) { int n = 0; for ( int i = 0; i < 4; n = ++i) { printf(" Values of ++i = %d\n", n); } int m = 0; for ( int j = 0; j < 4; m = j++) { printf(“ Values of j++ = %d\n", m); } return 0; } Values of ++i = 0 Values of ++i = 1 Values of ++i = 2 Values of ++i = 3 ------------------- Values of j++ = 0 Values of j++ = 0 Values of j++ = 1 Values of j++ = 2 Try to use “++i” and “++j” instead of the assignment “n” and “m”. You will get the same out ( which is “++i” ).
